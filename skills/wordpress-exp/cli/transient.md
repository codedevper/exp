## wp transient
wp transient get <key> [--format=<format>] -- get a transient value
wp transient set <key> <value> [<expiration>] -- set a transient
wp transient delete <key> -- delete a specific transient
wp transient delete-all [--expired] -- delete all transients (or just expired)
wp transient type -- show the transient storage type (database/object cache)
wp transient list [--search=<pattern>] [--site=<bool>] -- list transients
