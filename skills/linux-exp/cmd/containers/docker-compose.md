## docker-compose
sudo docker compose up -- create and start containers
sudo docker compose up -d -- start containers in background
sudo docker compose down -- stop and remove containers
sudo docker compose ps -- list containers
sudo docker compose logs -- view container logs
sudo docker compose logs -f -- follow logs
sudo docker compose build -- build or rebuild services
sudo docker compose pull -- pull service images
sudo docker compose start -- start services
sudo docker compose stop -- stop services
sudo docker compose restart -- restart services
sudo docker compose exec <service> <cmd> -- exec command in service
sudo docker compose config -- validate compose file
