## ping
ping <host> -- ping host continuously
ping -c <n> <host> -- send n packets
ping -i <s> <host> -- interval between packets (seconds)
ping -s <size> <host> -- specify packet size
ping -W <sec> <host> -- timeout in seconds
ping -4 <host> -- force IPv4
ping -6 <host> -- force IPv6
ping -D <host> -- print timestamps
ping -q <host> -- quiet output
ping -f <host> -- flood ping (root only)
