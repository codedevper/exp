## wp export
wp export [--post_type=<type>] [--post__in=<ids>] [--start_date=<date>] [--end_date=<date>] [--author=<login>] [--category=<slug>] [--status=<status>] -- export content to WXR
wp export --dir=<dir> -- save export files to a specific directory
wp export --filename_format=<format> -- custom filename format
wp export --skip_comments -- exclude comments from export
wp export --max_file_size=<MB> -- split export into multiple files at given size
