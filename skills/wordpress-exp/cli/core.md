## wp core
wp core download [--version=<version>] [--locale=<locale>] [--force] -- download WordPress core
wp core install --url=<url> --title=<title> --admin_user=<user> --admin_password=<pass> --admin_email=<email> -- run the install
wp core is-installed -- check if WordPress is installed
wp core version [--extra] -- show the WordPress version
wp core update [--version=<version>] [--force] -- update WordPress core
wp core update-db [--network] -- update the database after core update
wp core check-update -- check for core updates
wp core verify-checksums [--version=<version>] [--locale=<locale>] -- verify file checksums
wp core config -- alias for `wp config create`
