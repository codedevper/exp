## lsblk
lsblk -- list all block devices (tree view)
lsblk -f -- list with filesystem info
lsblk -m -- list with owner/group/mode
lsblk -l -- list in list format (no tree)
lsblk -o NAME,SIZE,TYPE,MOUNTPOINT -- custom columns
lsblk -d -- exclude dependent devices
lsblk -S -- show only SCSI devices
lsblk -p -- show full device paths
lsblk -a -- show all devices (including empty)
